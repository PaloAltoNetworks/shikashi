      def run(*args)
        t = args.pick(:timeout) do nil end
        binding_ = args.pick(Binding,:binding) do
          nil
        end

        begin
          timeout t do
            @evalhook_packet.run(binding_, @source, 0)
          end
        rescue ::Timeout::Error
          raise Shikashi::Timeout::Error
        end
      end
